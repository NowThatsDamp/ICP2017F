  function [OutVec] = convertTempWhile(InVec, x)
y = 1;
OutVec = zeros(1,length(InVec));
while  y <= length(InVec)
    if isequal(x, 'c2f')
    OutVec(y) = InVec(y)*(9/5)+32;
    y= y+1;
    elseif isequal(x, 'f2c')
    OutVec(y) = (InVec(y)-32)*(5/9);
    y=y+1;
    else
        error('Invalid Operation')
    end
end
end
